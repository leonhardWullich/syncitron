Testing

Dependency injection for fully testable sync

syncitron's architecture is fully testable via dependency injection. All core interfaces have in-memory or mock implementations.

Test Configuration

final config = syncitronConfig.testing();
// Minimal overhead: no logging, no metrics, fast timeouts

Test Doubles Strategy

InterfaceTest Double
LocalStoreIn-memory map-based implementation
RemoteAdapterFake with configurable responses
LoggerNoOpLogger() or mock that captures calls
MetricsCollectorInMemoryMetricsCollector or NoOpMetricsCollector

Example: In-Memory Local Store

class InMemoryLocalStore implements LocalStore {
  final Map<String, List<Map<String, dynamic>>> _tables = {};
  final Map<String, String> _cursors = {};

  List<Map<String, dynamic>> table(String name) =>
    _tables.putIfAbsent(name, () => []);

  @override
  Future<List<Map<String, dynamic>>> queryDirty(String table) async =>
    this.table(table).where((r) => r['is_synced'] == 0).toList();

  @override
  Future<void> upsertBatch(String table, List<Map<String, dynamic>> records) async {
    for (final record in records) {
      final idx = this.table(table).indexWhere((r) => r['id'] == record['id']);
      if (idx >= 0) this.table(table)[idx] = record;
      else this.table(table).add(record);
    }
  }
  // ... implement remaining LocalStore methods
}

Example: Fake Remote Adapter

class FakeRemoteAdapter extends RemoteAdapter {
  final Map<String, List<Map<String, dynamic>>> remoteTables = {};
  final List<Map<String, dynamic>> upserts = [];

  @override
  Future<PullResult> pull(PullRequest request) async {
    final records = remoteTables[request.table] ?? [];
    return PullResult(records: records, nextCursor: null);
  }

  @override
  Future<void> upsert({
    required String table,
    required Map<String, dynamic> data,
    String? idempotencyKey,
  }) async {
    upserts.add({'table': table, 'data': data, 'op_id': idempotencyKey});
  }

  // ... implement remaining RemoteAdapter methods
}

Integration Test

test('full sync cycle', () async {
  final local = InMemoryLocalStore();
  final remote = FakeRemoteAdapter();
  final engine = SyncEngine(
    localStore: local,
    remoteAdapter: remote,
    config: syncitronConfig.testing(),
    logger: NoOpLogger(),
  );

  engine.registerTable(TableConfig(
    name: 'todos',
    columns: ['id', 'title', 'updated_at', 'deleted_at'],
  ));

  await engine.init();
  final metrics = await engine.syncAll();
  expect(metrics.overallSuccess, isTrue);
});
