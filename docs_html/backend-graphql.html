GraphQL Backend

Works with Hasura, Apollo, or any GraphQL server

GraphQLAdapter works with any GraphQL server (Hasura, Apollo, Supabase GraphQL, custom). You provide query/mutation builders; syncitron handles the sync logic.

Setup

dependencies:
  syncitron: ^0.5.1
  graphql: ^5.1.0
import 'package:graphql/client.dart';
import 'package:syncitron/syncitron.dart';

final graphqlClient = GraphQLClient(
  link: HttpLink('https://your-server.com/graphql'),
  cache: GraphQLCache(),
);

final adapter = GraphQLAdapter(
  graphqlClient: graphqlClient,
  localStore: sqfliteStore,
  queryBuilder: (request) {
    return '''
      query Pull(\$cursor: timestamptz, \$limit: Int!) {
        ${request.table}(
          where: {updated_at: {_gte: \$cursor}}
          order_by: {updated_at: asc}
          limit: \$limit
        ) { ${request.columns.join(' ')} }
      }
    ''';
  },
  mutationBuilder: (table, data) {
    return '''
      mutation Upsert(\$object: ${table}_insert_input!) {
        insert_${table}_one(
          object: \$object
          on_conflict: {constraint: ${table}_pkey, update_columns: [${data.keys.join(', ')}]}
        ) { id }
      }
    ''';
  },
);

Real-Time via Subscriptions

final realtimeProvider = GraphQLRealtimeProvider(
  graphqlClient: graphqlClient,
  subscriptionQueryBuilder: (table) {
    return '''
      subscription Watch {
        $table(order_by: {updated_at: desc}, limit: 1) {
          id updated_at
        }
      }
    ''';
  },
);
